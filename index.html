Example Code #1: Simple Physics Calculation

# Example code to calculate velocity

initial_velocity = 0  # m/s
final_velocity = 20  # m/s
time = 5  # seconds

velocity = (final_velocity - initial_velocity) / time
print(f"Velocity: {velocity} m/s")

# Example code to solve a linear equation: ax + b = 0

# Using numpy to perform matrix multiplication

import numpy as np

A = np.array([[1, 2], [3, 4]])
